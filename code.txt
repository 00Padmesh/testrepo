    int priority;

{
    if(argc < 3)
    {
        printf("Not enough arguments\n");
        return 1;
    }

    char *description = malloc(20);
    if(description == NULL)
    {
        printf("Memory allocation failed\n");
        return 1;
    }

    userrequest *req = malloc(sizeof(userrequest));
    if(req == NULL)
    {
        printf("Memory allocation failed\n");
        free(description);
        return 1;
    }

    req->priority = 1;

    strncpy(req->name , "Guest" , sizeof(req->name) - 1);
    req->name[sizeof(req->name) - 1] = '\0';

    strncpy(description , argv[1] , 19);
    description[19] = '\0';

    strncpy(req->name , argv[2] , sizeof(req->name) - 1);
    req->name[sizeof(req->name) - 1] = '\0';

    printf("Description : %p\n" , description);
    printf("Req : %p\n" , req);
    printf("User %s\n" , req->name);
    printf("Priority level:%d\n" , req->priority);

    if(req->priority == 99)
    {
        printf("ADMIN ACCESS GRANTED\n");
    }

    free(description);
    free(req);

    return 0;
}

integer overflow

#include <stdio.h>
#include <limits.h>

int main() {

    int a = INT_MAX;   // maximum int value
    int b = 1;

    int result = a + b;   // OVERFLOW

    printf("INT_MAX = %d\n", a);
    printf("Result  = %d\n", result);

    return 0;
}


SAFE

#include <stdio.h>
#include <limits.h>

int main() {

    int a = INT_MAX;
    int b = 1;

    // CHECK BEFORE ADDITION
    if (a > INT_MAX - b) {
        printf("Integer overflow detected!\n");
    } else {
        int result = a + b;
        printf("Result = %d\n", result);
    }

    return 0;
}

    int count = INT_MAX / 2 + 10;   // very large number

    // INTEGER OVERFLOW happens here
    int total_size = count * sizeof(int);

    int *arr = (int *)malloc(total_size);

    if (arr == NULL) {
        printf("Allocation failed\n");

SECURE 

#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

int main() {

    int count = INT_MAX / 2 + 10;

    // SAFE CHECK BEFORE MULTIPLICATION
    if (count > INT_MAX / (int)sizeof(int)) {
        printf("Integer overflow detected!\n");
        return 0;
    }

    int total_size = count * sizeof(int);

    int *arr = (int *)malloc(total_size);

    if (arr == NULL) {
        printf("Allocation failed\n");
        return 0;
    }

    arr[0] = 100;
    printf("Value = %d\n", arr[0]);

    free(arr);
    return 0;
}
